<script>
    // Mantiene un conteo independiente para cada producto
    const counts = {
        count1: 0,
        count2: 0,
        count3: 0,
    };

    const min = 0;
    const max = 10;

    function increment(id) {
        if (counts[id] < max) {
            counts[id]++;
            updateDisplay(id);
            updateDatabase(counts[id]);
        }
    }

    function decrement(id) {
        if (counts[id] > min) {
            counts[id]--;
            updateDisplay(id);
            updateDatabase(counts[id]);
        }
    }

    function updateDisplay(id) {
        document.getElementById(id).innerText = counts[id];
    }

    function updateDatabase(value) {
        // Enviar el valor a un servidor para actualizar en la base de datos
        fetch('/update_cart', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'X-CSRFToken': '{{ csrf_token }}'  // Asegura la protecciÃ³n CSRF si usas Django
            },
            body: JSON.stringify({ itemCount: value }),
        })
        .then(response => response.json())
        .then(data => {
            console.log('Success:', data);
        })
        .catch((error) => {
            console.error('Error:', error);
        });
    }
</script>
